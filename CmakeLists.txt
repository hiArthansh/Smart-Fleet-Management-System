cmake_minimum_required(VERSION 3.10)

# Project Name
project(SmartFleetManagementSystem)

# Set C++ Standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Include header files
include_directories(src)

# Add executable
add_executable(SmartFleetManagementSystem
    src/main.cpp
    src/FleetManager.cpp
    src/Robot.cpp
    src/Grid.cpp
    src/PathPlanner.cpp
)
